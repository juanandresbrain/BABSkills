CREATE   VIEW dbo.stg_jumpmind_voids AS WITH /* ─────────────────────────────────────────────────────────────────────────    CTE 1: raw_voids    Identifies transactions where the cashier or back office requested a VOID.    Filters: transTypeCode = 'VOID' with both voidedDeviceId AND             voidedSequenceNumber populated (per Service1.cs:444-446).    ───────────────────────────────────────────────────────────────────────── */ raw_voids AS (     /* Source: jumpmind_sls_trans_summary (87 cols; carries trans_type_code,        voided_sequence_number, total, tax_total, customer fields).        Synthetic transaction_id per JumpMind composite-key convention        (device_id, business_date, sequence_number) — matches retail_line_item        and stg_jumpmind_transactions canonical_pos.        workstation_id derived from device_id; voided_device_id = device_id. */     SELECT         CAST(st.device_id        AS varchar(64)) + '|' +         CAST(st.business_date    AS varchar(8))  + '|' +         CAST(st.sequence_number  AS varchar(20))      AS void_transaction_id,         st.business_unit_id                           AS void_business_unit_id,         SUBSTRING(st.device_id,                   CHARINDEX('-', st.device_id) + 1,                   LEN(st.device_id))                  AS void_workstation_id,         st.sequence_number                            AS void_sequence_number,         st.create_time                                AS void_create_time,         st.barcode                                    AS void_barcode,         st.till_id                                    AS void_till_id,         st.username                                   AS void_username,         st.device_id                                  AS void_device_id,         st.business_date                              AS void_business_date,         /* voided_device_id = device_id (voids self-reference the same            register; no separate column on jumpmind_sls_trans_summary) */         st.device_id                                  AS voided_device_id,         st.voided_sequence_number,         st.trans_type_code       FROM LH_Source.dbo.jumpmind_sls_trans_summary AS st      WHERE st.trans_type_code = 'VOID'        AND st.voided_sequence_number IS NOT NULL ), /* ─────────────────────────────────────────────────────────────────────────    CTE 2: originals    Resolves each VOID to its original transaction by parsing voidedDeviceId.    Per Service1.cs:455-457:      var store = BuildStoreNumber(voidedDeviceId.Split('-')[0])      var voidedRegister = voidedDeviceId.Split('-')[1].TrimStart('0')    This is the OLD strip-leading-1 logic. Per BBW's 4-digit standardization,    we instead match on the FULL business_unit_id (no stripping).     For matching to LH_Source, the voided_device_id is in the form "<store>-<register>".    Match on:      - business_unit_id  ← derived from voided_device_id store portion      - workstation_id    ← derived from voided_device_id register portion      - sequence_number   ← voided_sequence_number    ───────────────────────────────────────────────────────────────────────── */ originals AS (     /* Original-transaction lookup: voids self-reference the same register, so        join on (device_id, voided_sequence_number) — no need to parse out a        separate voided_device_id (which doesn't exist anyway).        Money columns sourced from trans_summary:          gross_total = total          (net_total dropped May 7 — phantom column, no upstream definition           and no SmartLook report consumes it; per Ryan) */     SELECT         rv.void_transaction_id,         rv.void_business_unit_id,         rv.void_workstation_id,         rv.void_sequence_number,         rv.void_create_time,         rv.void_barcode,         rv.void_till_id,         rv.void_username,         rv.void_device_id,         rv.void_business_date,         CAST(st_orig.device_id       AS varchar(64)) + '|' +         CAST(st_orig.business_date   AS varchar(8))  + '|' +         CAST(st_orig.sequence_number AS varchar(20)) AS original_transaction_id,         st_orig.trans_type_code                       AS original_trans_type_code,         TRY_CAST(st_orig.business_date AS date)       AS original_transaction_date,         st_orig.total                                 AS original_gross_total,         rv.voided_device_id,         rv.voided_sequence_number       FROM raw_voids AS rv       LEFT JOIN LH_Source.dbo.jumpmind_sls_trans_summary AS st_orig         ON  st_orig.device_id        = rv.voided_device_id         AND st_orig.sequence_number  = rv.voided_sequence_number ) /* ─────────────────────────────────────────────────────────────────────────    Final SELECT: emit enriched POST_VOID records.    Per Service1.cs:464-475:      - Set transTypeCode = 'POST_VOID'      - Override sequenceNumber, deviceId, createTime, barcode, tillId, username        with the VOID record's values (the void itself's keys, not the original)      - All other fields come from the original      - voiding_reversal_flag stays 1 (BBW production convention)      - Tender-amount negation and price-modifier sequence rewrites happen in        stg_jumpmind_lines.sql / stg_canonical_payments.sql respectively, since        those touch the line-item and payment tables not exposed at this header        level. The flag column below signals downstream views to apply the        negation.    ───────────────────────────────────────────────────────────────────────── */ SELECT     o.void_transaction_id                              AS transaction_id,     o.void_business_unit_id                            AS business_unit_id,     o.void_workstation_id                              AS workstation_id,     o.void_sequence_number                             AS sequence_number,     o.void_create_time                                 AS create_time,     o.void_barcode                                     AS barcode,     o.void_till_id                                     AS till_id,     o.void_username                                    AS username,     o.void_device_id                                   AS device_id,     /* Relabeled type code per C#:471 */     CAST('POST_VOID' AS varchar(20))                   AS trans_type_code,     /* Original transaction's referenced sequence */     o.original_transaction_id                          AS referenced_original_transaction_id,     o.original_trans_type_code                         AS original_trans_type_code,     o.original_transaction_date                        AS original_transaction_date,     /* Per Service1.cs:482, voidedSequenceNumber is preserved as-is */     o.voided_sequence_number                           AS voided_sequence_number,     o.voided_device_id                                 AS voided_device_id,     /* Marker: tells stg_jumpmind_lines and stg_canonical_payments to negate        tender amounts and rewrite price modifier sequence numbers per        Service1.cs:485-499 */     CAST(1 AS bit)                                     AS void_enriched_flag   FROM originals AS o  WHERE o.original_transaction_id IS NOT NULL;  /* Only emit if original found */
