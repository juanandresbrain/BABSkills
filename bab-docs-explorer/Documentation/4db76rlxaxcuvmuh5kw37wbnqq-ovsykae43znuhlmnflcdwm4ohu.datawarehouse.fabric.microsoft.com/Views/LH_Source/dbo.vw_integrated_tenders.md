CREATE VIEW vw_integrated_tenders AS WITH pos_norm AS (   SELECT DISTINCT       CAST(tender_code                AS varchar(200))  AS tender_code,       CAST(tender_type_code           AS varchar(200))  AS tender_type_code,       CAST(COALESCE(iso_currency_code,'*') AS varchar(16)) AS iso_currency_code,       CAST([description]              AS varchar(400))  AS [description],       CAST(cash_drawer_open_required  AS bit)           AS cash_drawer_open_required,       CAST(till_unit_count_required   AS bit)           AS till_unit_count_required,  THEN 'American Express'         WHEN pproc LIKE 'GLOBAL%E' OR pproc='GLOBALE'                 THEN 'Global-E'         WHEN pproc = 'CASH'                                           THEN 'Cash'         WHEN pproc <> ''                                              THEN pproc         ELSE 'Other'       END AS tender_code,       CASE         WHEN psub LIKE '%APPLEPAY%'  OR pproc LIKE 'ADYEN_APPLEPAY%'  THEN 'E-Wallet'         WHEN psub LIKE '%GOOGLEPAY%' OR pproc LIKE 'ADYEN_GOOGLEPAY%' THEN 'E-Wallet'         WHEN psub LIKE '%PAYPAL%'    OR pproc LIKE 'ADYEN_PAYPAL%'    THEN 'E-Wallet'         WHEN psub LIKE '%GIFTCARD%'  OR pproc LIKE 'ADYEN_GIFTCARD%'  THEN 'E-Wallet'         WHEN psub LIKE '%KLARNA%'    OR pproc LIKE 'ADYEN_KLARNA%'    THEN 'E-Wallet'         WHEN psub LIKE '%AFTERPAY%'  OR THEN 'Cash'         WHEN pproc = 'CYBERSOURCE'                                    THEN 'Cybersource'         WHEN pproc LIKE 'ADYEN%'                                      THEN 'E-Wallet'         WHEN CAST('*' AS varchar(16)) AS iso_currency_code,       CASE         WHEN psub LIKE '%APPLEPAY%'  OR pproc LIKE 'ADYEN_APPLEPAY%'  THEN 'ADYEN_APPLEPAY'         WHEN psub LIKE '%GOOGLEPAY%' OR pproc LIKE 'ADYEN_GOOGLEPAY%' THEN 'ADYEN_GOOGLEPAY'         WHEN psub LIKE '%PAYPAL%'    OR pproc LIKE 'ADYEN_PAYPAL%'    THEN 'ADYEN_PAYPAL'         WHEN psub LIKE '%GIFTCARD%'  OR pproc LIKE 'ADYEN_GIFTCARD%' 'ADYEN_AFTERPAY'         WHEN pproc LIKE 'GLOBAL%E'   OR pproc='GLOBALE'               THEN 'GLOBAL_E'         WHEN pproc = 'CASH'                                           THEN 'CASH'         WHEN
